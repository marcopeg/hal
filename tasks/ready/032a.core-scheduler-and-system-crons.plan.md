### Step 2.2 — `src/crons/schema.ts`

Zod schema for `.md` cron frontmatter (system tier — `projectId` is required in targets).

```ts
import { z } from "zod";

export const CronTargetSchema = z.object({
  projectId: z.string().min(1),
  userId: z.number().int().optional(),
  flowResult: z.boolean().optional(),
});

export const MdFrontmatterSchema = z
  .object({
    name: z.string().optional(),
    enabled: z.boolean().default(true),
    schedule: z.string().optional(),
    runAt: z.string().optional(),  // ISO 8601; parsed to Date in loader
    targets: z.array(CronTargetSchema).min(1),
  })
  .refine(
    (data) => !!(data.schedule || data.runAt),
    { message: "Exactly one of schedule or runAt is required" },
  )
  .refine(
    (data) => !(data.schedule && data.runAt),
    { message: "Only one of schedule or runAt may be set" },
  );

export type MdFrontmatter = z.infer<typeof MdFrontmatterSchema>;
```

Validate `flowResult: true` requires `userId` in the loader (not in schema, to allow better error messages).

## Phase 3 — Loaders

### Step 3.1 — `src/crons/loader-md.ts`

```ts
import { readFileSync } from "node:fs";
import { basename } from "node:path";
import { parse as parseYaml } from "yaml";
import { MdFrontmatterSchema } from "./schema.js";
import type { MdCronDefinition } from "./types.js";

/**
 * Parse a .md cron file.
 * Throws a descriptive error if the file is invalid.
 * @param strict - if true, flowResult+userId mismatch throws; if false, logs only
 */
export function loadMdCron(
  filePath: string,
  options: { strict: boolean },
): MdCronDefinition {
  const raw = readFileSync(filePath, "utf-8");

  // Split frontmatter from body
  const match = raw.match(/^---\n([\s\S]*?)\n---\n?([\s\S]*)$/);
  if (!match) throw new Error(`No YAML frontmatter found in ${filePath}`);

  const [, frontmatterRaw, body] = match;
  const parsed = parseYaml(frontmatterRaw);
  const result = MdFrontmatterSchema.safeParse(parsed);
  if (!result.success) {
    throw new Error(`Invalid frontmatter in ${filePath}: ${result.error.message}`);
  }

  const fm = result.data;

  // Validate flowResult+userId co-presence
  for (const target of fm.targets) {
    if (target.flowResult && !target.userId) {
      const msg = `flowResult: true requires userId in targets (${filePath})`;
      if (options.strict) throw new Error(msg);
      // caller handles non-strict logging
      throw Object.assign(new Error(msg), { soft: true });
    }
  }

  const nameFromFile = basename(filePath, ".md");

  return {
    type: "md",
    name: fm.name ?? nameFromFile,
    sourceFile: filePath,
    schedule: fm.schedule,
    runAt: fm.runAt ? new Date(fm.runAt) : undefined,
    enabled: fm.enabled,
    targets: fm.targets,
    prompt: body.trim(),
  };
}
```

### Step 3.2 — `src/crons/loader-mjs.ts`

```ts
import { basename } from "node:path";
import type { MjsCronDefinition, CronContext } from "./types.js";

/**
 * Dynamically import a .mjs cron module and validate its exports.
 * Cache-busts with ?t= to pick up changes on hot reload (matching existing pattern).
 * Throws a descriptive error if exports are invalid.
 */
export async function loadMjsCron(filePath: string): Promise<MjsCronDefinition> {
  const url = `${filePath}?t=${Date.now()}`;
  const mod = await import(url);

  if (typeof mod.handler !== "function") {
    throw new Error(`Missing exported handler() in ${filePath}`);
  }
  if (!mod.schedule && !mod.runAt) {
    throw new Error(`Missing schedule or runAt export in ${filePath}`);
  }
  if (mod.schedule && mod.runAt) {
    throw new Error(`Only one of schedule or runAt may be exported in ${filePath}`);
  }

  const nameFromFile = basename(filePath, ".mjs");

  return {
    type: "mjs",
    name: mod.name ?? nameFromFile,
    sourceFile: filePath,
    schedule: typeof mod.schedule === "string" ? mod.schedule : undefined,
    runAt: mod.runAt ? new Date(mod.runAt) : undefined,
    handler: mod.handler as (ctx: CronContext) => Promise<void>,
  };
}
```

### Step 3.3 — `src/crons/loader-dir.ts`

```ts
import { readdirSync } from "node:fs";
import { join } from "node:path";
import type pino from "pino";
import { loadMdCron } from "./loader-md.js";
import { loadMjsCron } from "./loader-mjs.js";
import type { CronDefinition } from "./types.js";

export interface LoadDirResult {
  definitions: CronDefinition[];
  errors: Array<{ file: string; error: string }>;
}

/**
 * Load all .md and .mjs cron files from a directory.
 * @param strict - if true, throw on first error (boot); if false, collect errors (hot reload)
 */
export async function loadCronsFromDir(
  dir: string,
  logger: pino.Logger,
  options: { strict: boolean },
): Promise<LoadDirResult> {
  let files: string[];
  try {
    files = readdirSync(dir).filter(
      (f) => f.endsWith(".md") || f.endsWith(".mjs"),
    );
  } catch {
    // Directory does not exist — not an error, just no crons
    return { definitions: [], errors: [] };
  }

  const definitions: CronDefinition[] = [];
  const errors: Array<{ file: string; error: string }> = [];

  for (const file of files) {
    const filePath = join(dir, file);
    try {
      if (file.endsWith(".md")) {
        definitions.push(loadMdCron(filePath, { strict: options.strict }));
      } else {
        definitions.push(await loadMjsCron(filePath));
      }
    } catch (err) {
      const message = err instanceof Error ? err.message : String(err);
      if (options.strict) throw new Error(message);
      errors.push({ file: filePath, error: message });
      logger.error({ file: filePath, error: message }, "Cron file invalid — skipping");
    }
  }

  return { definitions, errors };
}
```

### Step 4.1 — `src/crons/log.ts`

```ts
import { mkdirSync, appendFileSync } from "node:fs";
import { join } from "node:path";

export interface CronLogEntry {
  jobName: string;
  sourceFile: string;
  startedAt: Date;
  finishedAt: Date;
  output: string;
  error?: string;
}

/**
 * Write a cron execution log to:
 *   {logBaseDir}/.hal/logs/crons/{jobName}/{timestamp}.{jobName}.txt
 */
export function writeCronLog(logBaseDir: string, entry: CronLogEntry): void {
  const logDir = join(logBaseDir, ".hal", "logs", "crons", entry.jobName);
  mkdirSync(logDir, { recursive: true });

  const ts = entry.startedAt
    .toISOString()
    .replace(/[:.]/g, "-")
    .replace("Z", "");
  const filename = `${ts}.${entry.jobName}.txt`;
  const filePath = join(logDir, filename);

  const lines = [
    `job:     ${entry.jobName}`,
    `source:  ${entry.sourceFile}`,
    `started: ${entry.startedAt.toISOString()}`,
    `ended:   ${entry.finishedAt.toISOString()}`,
    `status:  ${entry.error ? "error" : "ok"}`,
    "",
    "--- output ---",
    entry.output,
  ];
  if (entry.error) {
    lines.push("", "--- error ---", entry.error);
  }

  appendFileSync(filePath, lines.join("\n") + "\n", "utf-8");
}
```

### Step 4.2 — `src/crons/executor-md.ts`

```ts
import type pino from "pino";
import type { ProjectContext } from "../types.js";
import { createAgent } from "../agent/index.js";
import { writeCronLog } from "./log.js";
import type { MdCronDefinition, CronContext } from "./types.js";

/**
 * Execute a .md cron definition.
 * Uses the project's engine to call the prompt, sends result if flowResult: true.
 *
 * @param internalProjectCtxs - full ProjectContext per project slug (not exposed in public CronContext)
 * @param logBaseDir - base directory for writing execution logs (configDir for system tier)
 */
export async function executeMdCron(
  def: MdCronDefinition,
  internalProjectCtxs: Record<string, ProjectContext>,
  cronCtx: CronContext,
  logBaseDir: string,
  logger: pino.Logger,
): Promise<void> {
  for (const target of def.targets) {
    const projectCtx = internalProjectCtxs[target.projectId];
    if (!projectCtx) {
      logger.error(
        { jobName: def.name, projectId: target.projectId },
        "Cron target projectId not found — skipping target",
      );
      continue;
    }

    const startedAt = new Date();
    let output = "";
    let error: string | undefined;

    try {
      const agent = createAgent(projectCtx);
      output = await agent.call(def.prompt);

      // Send result via Telegram DM if requested
      if (target.flowResult && target.userId) {
        const projectCronCtx = cronCtx.projects[target.projectId];
        await projectCronCtx.bot.api.sendMessage(target.userId, output);
      }
    } catch (err) {
      error = err instanceof Error ? err.message : String(err);
      logger.error(
        { jobName: def.name, projectId: target.projectId, error },
        "Cron .md execution failed",
      );
    }

    writeCronLog(logBaseDir, {
      jobName: def.name,
      sourceFile: def.sourceFile,
      startedAt,
      finishedAt: new Date(),
      output,
      error,
    });
  }
}
```

### Step 5.1 — `src/crons/context.ts`

```ts
import type { Bot } from "grammy";
import type { ProjectContext } from "../types.js";
import type { CronContext } from "./types.js";

export interface CronRuntimeEntry {
  projectCtx: ProjectContext;
  bot: Bot;
}

/**
 * Build the public CronContext and the internal projectCtxs map for agent execution.
 * The public CronContext is passed to .mjs handlers.
 * The internal map is used by .md executors to call createAgent().
 */
export function buildCronContext(
  multiConfig: Record<string, unknown>,
  entries: Record<string, CronRuntimeEntry>,
): {
  cronCtx: CronContext;
  internalProjectCtxs: Record<string, ProjectContext>;
} {
  const projects: CronContext["projects"] = {};
  const internalProjectCtxs: Record<string, ProjectContext> = {};

  for (const [slug, { projectCtx, bot }] of Object.entries(entries)) {
    projects[slug] = { config: projectCtx.config, bot };
    internalProjectCtxs[slug] = projectCtx;
  }

  return {
    cronCtx: { config: multiConfig, projects },
    internalProjectCtxs,
  };
}
```

### Step 5.2 — `src/crons/scheduler.ts`

```ts
import { Cron } from "croner";
import type pino from "pino";
import { executeMdCron } from "./executor-md.js";
import { executeMjsCron } from "./executor-mjs.js";
import type {
  CronDefinition,
  CronContext,
} from "./types.js";
import type { ProjectContext } from "../types.js";

interface JobEntry {
  definition: CronDefinition;
  cronInstance: Cron | null;  // null if skipped (past runAt)
}

export class CronScheduler {
  private jobs = new Map<string, JobEntry>();

  constructor(
    private readonly cronCtx: CronContext,
    private readonly internalProjectCtxs: Record<string, ProjectContext>,
    private readonly logBaseDir: string,
    private readonly logger: pino.Logger,
  ) {}

  /** Load an array of definitions and schedule all eligible jobs. */
  load(definitions: CronDefinition[]): void {
    for (const def of definitions) {
      this.add(def);
    }
  }

  /** Add and schedule a single job. Replaces existing job with the same name. */
  add(def: CronDefinition): void {
    this.remove(def.name);

    if ((def as { enabled?: boolean }).enabled === false) {
      this.logger.debug({ jobName: def.name }, "Cron disabled — not scheduled");
      return;
    }

    if (def.runAt) {
      if (def.runAt <= new Date()) {
        this.logger.debug(
          { jobName: def.name, runAt: def.runAt.toISOString() },
          "Cron runAt is in the past — skipping",
        );
        this.jobs.set(def.name, { definition: def, cronInstance: null });
        return;
      }
    }

    const pattern = def.runAt
      ? def.runAt  // croner accepts Date for one-off
      : def.schedule!;

    const cronInstance = new Cron(pattern, { protect: true }, async () => {
      this.logger.info({ jobName: def.name }, "Cron firing");
      await this.execute(def);
      // One-off: croner stops automatically when Date trigger fires once.
    });

    this.jobs.set(def.name, { definition: def, cronInstance });
    this.logger.info(
      { jobName: def.name, pattern: def.runAt?.toISOString() ?? def.schedule },
      "Cron scheduled",
    );
  }

  /** Remove and stop a job by name. No-op if not found. */
  remove(name: string): void {
    const entry = this.jobs.get(name);
    if (entry?.cronInstance) {
      entry.cronInstance.stop();
    }
    this.jobs.delete(name);
  }

  /** Replace a job (used by file watcher on change). */
  replace(def: CronDefinition): void {
    this.add(def);  // add() already calls remove() first
  }

  /** Stop all scheduled jobs. */
  stop(): void {
    for (const [name, entry] of this.jobs) {
      if (entry.cronInstance) {
        entry.cronInstance.stop();
        this.logger.debug({ jobName: name }, "Cron stopped");
      }
    }
    this.jobs.clear();
  }

  private async execute(def: CronDefinition): Promise<void> {
    try {
      if (def.type === "md") {
        await executeMdCron(
          def,
          this.internalProjectCtxs,
          this.cronCtx,
          this.logBaseDir,
          this.logger,
        );
      } else {
        await executeMjsCron(def, this.cronCtx, this.logBaseDir, this.logger);
      }
    } catch (err) {
      this.logger.error(
        { jobName: def.name, error: err instanceof Error ? err.message : String(err) },
        "Unhandled error in cron execution",
      );
    }
  }
}
```

> **Implementation note on croner API:** Verify that `new Cron(date, options, handler)` is the correct signature for one-off Date triggers in the installed croner version. Adjust if the API differs.

## Phase 6 — File watcher and public API

### Step 6.1 — `src/crons/watcher.ts`

Follows the same pattern as `src/bot/commands/watcher.ts`.

```ts
import { mkdirSync } from "node:fs";
import type pino from "pino";
import { loadMdCron } from "./loader-md.js";
import { loadMjsCron } from "./loader-mjs.js";
import type { CronScheduler } from "./scheduler.js";

export interface CronWatcher {
  stop: () => Promise<void>;
}

export function startCronWatcher(
  cronDir: string,
  scheduler: CronScheduler,
  logger: pino.Logger,
): CronWatcher {
  mkdirSync(cronDir, { recursive: true });

  let debounceTimers = new Map<string, ReturnType<typeof setTimeout>>();
  let watcherInstance: { close: () => Promise<void> } | null = null;

  const DEBOUNCE_MS = 300;

  function scheduleHandle(filePath: string, event: "add" | "change" | "unlink"): void {
    const existing = debounceTimers.get(filePath);
    if (existing) clearTimeout(existing);
    debounceTimers.set(
      filePath,
      setTimeout(() => {
        debounceTimers.delete(filePath);
        void handleFile(filePath, event);
      }, DEBOUNCE_MS),
    );
  }

  async function handleFile(filePath: string, event: "add" | "change" | "unlink"): Promise<void> {
    const isRelevant = filePath.endsWith(".md") || filePath.endsWith(".mjs");
    if (!isRelevant) return;

    logger.debug({ filePath, event }, "Cron file event");

    if (event === "unlink") {
      // Derive job name from file path (filename without extension)
      const name = filePath.split("/").pop()!.replace(/\.(md|mjs)$/, "");
      scheduler.remove(name);
      logger.info({ jobName: name }, "Cron removed (file deleted)");
      return;
    }

    try {
      const def = filePath.endsWith(".md")
        ? loadMdCron(filePath, { strict: false })
        : await loadMjsCron(filePath);

      if (event === "add") {
        scheduler.add(def);
        logger.info({ jobName: def.name }, "Cron added (file created)");
      } else {
        scheduler.replace(def);
        logger.info({ jobName: def.name }, "Cron updated (file changed)");
      }
    } catch (err) {
      logger.error(
        { filePath, error: err instanceof Error ? err.message : String(err) },
        "Cron hot-reload: invalid file — skipping",
      );
    }
  }

  const watcherReady = (async () => {
    try {
      const chokidar = await import("chokidar");
      const watcher = chokidar.watch(cronDir, {
        ignoreInitial: true,
        persistent: true,
      });

      watcher.on("add", (p: string) => scheduleHandle(p, "add"));
      watcher.on("change", (p: string) => scheduleHandle(p, "change"));
      watcher.on("unlink", (p: string) => scheduleHandle(p, "unlink"));
      watcher.on("error", (err: unknown) => {
        logger.error(
          { error: err instanceof Error ? err.message : String(err) },
          "Cron watcher error",
        );
      });

      watcherInstance = watcher;
      logger.debug({ cronDir }, "Cron file watcher started");
    } catch (err) {
      logger.error(
        { error: err instanceof Error ? err.message : String(err) },
        "Failed to start cron file watcher",
      );
    }
  })();

  return {
    stop: async () => {
      for (const timer of debounceTimers.values()) clearTimeout(timer);
      debounceTimers.clear();
      await watcherReady;
      if (watcherInstance) {
        await watcherInstance.close();
        watcherInstance = null;
        logger.debug("Cron file watcher stopped");
      }
    },
  };
}
```

### Step 6.2 — `src/crons/index.ts`

```ts
import { join } from "node:path";
import type pino from "pino";
import type { Bot } from "grammy";
import type { ProjectContext } from "../types.js";
import { loadCronsFromDir } from "./loader-dir.js";
import { buildCronContext } from "./context.js";
import { CronScheduler } from "./scheduler.js";
import { startCronWatcher } from "./watcher.js";
import type { CronHandle } from "./types.js";

export interface StartSystemCronsOptions {
  configDir: string;
  multiConfig: Record<string, unknown>;
  /** Map of project slug → { projectCtx, bot } */
  projects: Record<string, { projectCtx: ProjectContext; bot: Bot }>;
  logger: pino.Logger;
}

/**
 * Load, validate, and schedule all system-level cron files from
 * {configDir}/.hal/crons/. Starts a file watcher for hot reload.
 *
 * Throws on the first invalid file at boot (strict mode).
 * Returns a handle to stop the scheduler and watcher.
 */
export async function startSystemCrons(options: StartSystemCronsOptions): Promise<CronHandle> {
  const { configDir, multiConfig, projects, logger } = options;
  const cronDir = join(configDir, ".hal", "crons");
  const logBaseDir = configDir;

  const { cronCtx, internalProjectCtxs } = buildCronContext(multiConfig, projects);

  // Load all cron files — strict: true at boot
  const { definitions } = await loadCronsFromDir(cronDir, logger, { strict: true });
  logger.info({ count: definitions.length }, "System crons loaded");

  const scheduler = new CronScheduler(cronCtx, internalProjectCtxs, logBaseDir, logger);
  scheduler.load(definitions);

  const watcher = startCronWatcher(cronDir, scheduler, logger);

  return {
    stop: async () => {
      await watcher.stop();
      scheduler.stop();
      logger.info("System cron scheduler stopped");
    },
  };
}

// Build the projects map for the cron runtime
const cronProjects: Record<string, { projectCtx: ProjectContext; bot: Bot }> = {};
for (let i = 0; i < contexts.length; i++) {
  const ctx = contexts[i];
  cronProjects[ctx.config.slug] = {
    projectCtx: ctx,
    bot: botHandles[i].bot,
  };
}

// Start system crons (strict: throws on first invalid file)
const cronHandle = await startSystemCrons({
  configDir,
  multiConfig: loaded.config as Record<string, unknown>,
  projects: cronProjects,
  logger: startupLogger,
});
```

Return `cronHandle` from `runBotsForConfig()` alongside `botHandles`:

```ts
interface RunResult {
  botHandles: BotHandle[];
  cronHandle: CronHandle;
}
```

### Step 7.3 — Wire cron shutdown in `cli.ts`

In the config-watcher reload callback (stop → reload → restart):
```ts
// Stop crons alongside bots on reload
await state.cronHandle?.stop();
// ... reload config, restart bots ...
state.cronHandle = nextResult.cronHandle;
```

In the shutdown handler:
```ts
async function shutdown(signal: string): Promise<void> {
  await configWatcher.stop();
  await state.cronHandle?.stop();
  await Promise.all(state.botHandles.map((h) => h.stop().catch(() => {})));
  process.exit(0);
}
```
